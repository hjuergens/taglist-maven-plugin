44  /**
45   * Class that analyzes a file with a special comment tag. For instance:
46   * 
47   * <pre>
48   * // TODO: Example of an Eclipse/IntelliJ-like "todo" tag
49   * </pre>
50   * 
51   * @author <a href="mailto:bellingard.NO-SPAM@gmail.com">Fabrice Bellingard </a>
52   * @todo : This is another example of "todo" tag
53   */
54  public class FileAnalyser
55  {
56      /**
57       * String that is used for beginning a comment line.
58       */
59      private static final String STAR_COMMENT = "*";
60  
61      /**
62       * String that is used for beginning a comment line.
63       */
64      private static final String SLASH_COMMENT = "//";
65      
66      /**
67       * Maximum length of a comment.
68       */
69      private static final int MAX_COMMENT_CHARACTERS = 99999;
70  
71      /**
72       * The character encoding of the files to analyze.
73       */
74      private String encoding;
75      
76      /**
77       * The Locale of the files to analyze.
78       */
79      private Locale locale;
80  
81      /**
82       * The directories to analyze.
83       */
84      private Collection sourceDirs;
85  
86      /**
87       * Log for debug output.
88       */
89      private Log log;
90  
91      /**
92       * Set to true if the analyzer should look for multiple line comments.
93       */
94      private boolean multipleLineCommentsOn;
95  
96      /**
97       * Set to true if the analyzer should look for tags without comments.
98       */
99      private boolean emptyCommentsOn;
100 
101     /**
102      * String used to indicate that there is no comment after the tag.
103      */
104     private String noCommentString;

106     /**
107      * ArrayList of tag classes.

109     private ArrayList tagClasses = new ArrayList();

111     /**
112      * Constructor.
113      * 
114      * @param report the MOJO that is using this analyzer.
115      * @param tagClasses the array of tag classes to use for searching
116      */
117     public FileAnalyser( TagListReport report, ArrayList tagClasses )
118     {
119         multipleLineCommentsOn = report.isMultipleLineComments();
120         emptyCommentsOn = report.isEmptyComments();
121         log = report.getLog();
122         sourceDirs = report.constructSourceDirs();
123         encoding = report.getEncoding();
124         locale = report.getLocale();
125         noCommentString = report.getBundle().getString( "report.taglist.nocomment" );      
126         this.tagClasses = tagClasses;
127     }

129     /**
130      * Execute the analysis for the configuration given by the TagListReport.
131      * 
132      * @return a collection of TagReport objects.
133      * @throws MavenReportException the Maven report exception.
134      */
135     public Collection execute()
136         throws MavenReportException
137     {
138         List fileList = findFilesToScan();
139 
140         for ( Iterator iter = fileList.iterator(); iter.hasNext(); )
141         {
142             File file = (File) iter.next();
143             if ( file.exists() )
144             {
145                 scanFile( file );
146             }
147         }
148 
149         // Get the tag reports from each of the tag classes.
150         Collection tagReports = new ArrayList();
151         Iterator itr = tagClasses.iterator();      
152         while ( itr.hasNext() )
153         {
154             TagClass tc = (TagClass) itr.next();
155             tagReports.add( tc.getTagReport() );
156         }
157 
158         return tagReports;
159     }
160     /**
161      * Gives the list of files to scan.
162      * 
163      * @return a List of File objects.
164      * @throws MavenReportException the Maven report exception.
165      */
166     private List findFilesToScan()
167         throws MavenReportException
168     {
169         List filesList = new ArrayList();
170         try
171         {
172             for ( Iterator iter = sourceDirs.iterator(); iter.hasNext(); )
173             {
174                 filesList.addAll( FileUtils.getFiles( new File( (String) iter.next() ), "**/*.java", null ) );
175             }
176         }
177         catch ( IOException e )
178         {
179             throw new MavenReportException( "Error while trying to find the files to scan.", e );
180         }
181         return filesList;
182     }

184     /**
185      * Access an input reader that uses the current file encoding.
186      *
187      * @param file the file to open in the reader.
188      * @throws IOException the IO exception.
189      * @return a reader with the current file encoding.
190      */
191     private Reader getReader( File file ) throws IOException
192     {
193         InputStream in = new FileInputStream( file );
194         return ( encoding == null ) ? new InputStreamReader( in ) : new InputStreamReader( in, encoding ); 
195     }

197     /**
198      * Scans a file to look for task tags.
199      * 
200      * @param file the file to scan.
201      */
202     public void scanFile( File file )
203     {
204         LineNumberReader reader = null;
205 
206         try
207         {
208             reader = new LineNumberReader( getReader( file ) );
209 
210             String currentLine = reader.readLine();
211             while ( currentLine != null )
212             {
213                 int index = -1;
214                 Iterator iter = tagClasses.iterator();
215                 // look for a tag on this line
216                 while ( iter.hasNext() )
217                 {
218                     TagClass tagClass = (TagClass) iter.next();
219                     index = tagClass.tagMatchContains( currentLine, locale );
220                     if ( index != TagClass.NO_MATCH )
221                     {
222                         // there's a tag on this line
223                         String commentType = null;
224                         commentType = extractCommentType( currentLine, index );
225 
226                         if ( commentType == null )
227                         {
228                             // this is not a valid comment tag: skip other tag classes and
229                             // go to the next line
230                             break;
231                         }
232 
233                         int tagLength = tagClass.getLastTagMatchStringLength();
234                         int commentStartIndex = reader.getLineNumber();
235                         StringBuffer comment = new StringBuffer();
236 
237                         String firstLine = StringUtils.strip( currentLine.substring( index + tagLength ) );
238                         firstLine = StringUtils.removeEnd( firstLine, "*/" ); //MTAGLIST-35
239                         if ( firstLine.length() == 0 || ":".equals( firstLine ) )
240                         {
241                             // this is not a valid comment tag: nothing is written there
242                             if ( emptyCommentsOn )
243                             {
244                                 comment.append( "--" );
245                                 comment.append( noCommentString );
246                                 comment.append( "--" );
247                             }
248                             else
249                             {                               
250                                 continue;
251                             }
252                         }
253                         else
254                         {
255                             // this tag has a comment
256                             if ( firstLine.charAt( 0 ) == ':' )
257                             {
258                                 comment.append( firstLine.substring( 1 ).trim() );
259                             }
260                             else
261                             {
262                                 comment.append( firstLine );
263                             }
264 
265                             if ( multipleLineCommentsOn )
266                             {
267                                 // Mark the current position, set the read forward limit to
268                                 // a large number that should not be met.
269                                 reader.mark( MAX_COMMENT_CHARACTERS );
270                                 
271                                 // next line
272                                 String futureLine = reader.readLine();
273                                 
274                                 // we're looking for multiple line comments
275                                 while ( futureLine != null && futureLine.trim().startsWith( commentType )
276                                     && futureLine.indexOf( tagClass.getLastTagMatchString() ) < 0 )
277                                 {
278                                     String currentComment = futureLine.substring( futureLine.indexOf( commentType )
279                                                                                    + commentType.length() ).trim();
280                                     if ( currentComment.startsWith( "@" ) || "".equals( currentComment )
281                                         || "/".equals( currentComment ) )
282                                     {
283                                         // the comment is finished
284                                         break;
285                                     }
286                                     // try to look if the next line is not a new tag
287                                     boolean newTagFound = false;
288                                     Iterator moreTCiter = tagClasses.iterator();
289                                     while ( moreTCiter.hasNext() )
290                                     {
291                                         TagClass tc = (TagClass) moreTCiter.next();
292                                         if ( tc.tagMatchStartsWith( currentComment, locale ) )
293                                         {
294                                             newTagFound = true;
295                                             break;
296                                         }
297                                     }
298                                     if ( newTagFound )
299                                     {
300                                         // this is a new comment: stop here the current comment
301                                         break;
302                                     }
303                                     // nothing was found: this means the comment is going on this line
304                                     comment.append( " " );
305                                     comment.append( currentComment );
306                                     futureLine = reader.readLine();
307                                 }
308                                 
309                                 // Reset the reader to the marked position before the multi
310                                 // line check was performed.
311                                 reader.reset();
312                             }
313                         }
314                         TagReport tagReport = tagClass.getTagReport();
315                         FileReport fileReport = tagReport.getFileReport( file, encoding );
316                         fileReport.addComment( comment.toString(), commentStartIndex );
317                     }
318                 }
319                 currentLine = reader.readLine();
320             }
321         }
322         catch ( IOException e )
323         {
324             log.error( "Error while scanning the file " + file.getPath(), e );
325         }
326         finally
327         {
328             IOUtil.close( reader );
329         }
330     }

332     /**
333      * Finds the type of comment the tag is in.
334      * 
335      * @param currentLine the line to analyze.
336      * @param index the index of the tag in the line.
337      * @return "*" or "//" or null.
338      */
339     private String extractCommentType( String currentLine, int index )
340     {
341         String commentType = null;
342         String beforeTag = currentLine.substring( 0, index ).trim();
343         if ( beforeTag.endsWith( SLASH_COMMENT ) )
344         {
345             commentType = SLASH_COMMENT;
346         }
347         else if ( beforeTag.endsWith( STAR_COMMENT ) )
348         {
349             commentType = STAR_COMMENT;
350         }
351         return commentType;
352     }
